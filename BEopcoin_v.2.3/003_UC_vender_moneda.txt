Modulo: Venta Moneda

POST:/api/transacion/vender

DTO
(request)
{
  CuentaDebito (ID_Billetera, Tipo Moneda: ANY != Peso),
  CuentaCredito (ID_Billetera, TipoMoneda: Peso),
  Importe (TipoMoneda = TipoMoneda CuentaDebito)
}

DTO
(response)
{
  SaldoCuentaDebito (En el TipoMoneda vendido),
  SaldoCuentaCredito (en Pesos),
  ImporteVendidoEnTipoMoneda (En el TipoMoneda vendido),
  FechaTransacion (fecha en formato DateTime)	
}
